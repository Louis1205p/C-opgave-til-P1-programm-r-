using System;

class Program
{
    static void Main()
    {
        int i = 1;  // Initialiserer tælleren og gir den værdien 1

        while (i <= 10)
        {
            Console.WriteLine(i);  // Udskriver den nuværende værdi af i som er en
            i++;  
        }
    }
}
